pkg load signal;
pkg load symbolic;
pkg load image;
pkg load nnet;

# Diretório de treinamento
cd inst

# Funções de treinamento
test __analyzerows
test __copycoltopos1

test isposint
test min_max
test newff

# Retorna à pasta anterior
cd ..

%%Rede neural
deepnet = nnet;

%%Imagem
webcamlist;
try
  cam = webcam;
end

%%Classificação
img = snapshot(cam);
img = imresize(img,[227,227]);
pred1 = classify(deepnet,img);
figure;
  imshow(img,[]);
  pred1;
