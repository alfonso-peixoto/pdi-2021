pkg load signal;

pkg load image;
img = im2double(imread("benign (1).png"));
figure;
imshow(img);

level = graythresh(img);
BW = im2bw(img,0.3);
figure;
imshow(BW);

bwAreaOpen = imcomplement(bwareaopen(BW,2000));
strelname = strel("disk",8,0);
figure;
  imshow(bwAreaOpen,[]);
  bwBordas = imclearborder(bwAreaOpen, 8);
figure; imshow(bwBordas);
  imTumor = imopen(bwBordas, strelname); % formatos de disco não implementados: error: strel: N for disk shape not yet implemented, use N of 0
