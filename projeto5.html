Carregar a imagem

img = im2double(imread("Fig0507(a)(ckt-board-orig).tif"));
imshow(img);
title("Figura original");

(1) Reescalar a imagem

for i = 1:4
  redimension = imresize(img,i);
  titleName = sprintf("Escala k=%d",i);
  figure;
    imshow(redimension);
    title(titleName);
end

(2) Aplicar ruído imnoise (sal e pimenta)

imgNoisy = imnoise(img,'salt & pepper',0.2);
figure;
  imshow(imgNoisy);
  title("Imagem com ruído de sal e pimenta");

(3) Aplicar o filtro de mediana

maximum = 9;
quantity = 5;
position = 1;
subplot(1,quantity,position);
  imshow(imgNoisy);
  title("Imagem ruidosa");
for i = 3:2:maximum
  position = position + 1;
  img = mat2gray(medfilt2(imgNoisy,[i,i], "symmetric"));
  titulo = sprintf("Filtro de ordem %d",i);
  subplot(1, quantity, position);
    imshow(img);
    title(titleName);
end
